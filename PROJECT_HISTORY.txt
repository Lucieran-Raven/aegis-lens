## 2.1 Pipeline 1: Frame-Timing Entropy (CHRONOS)

### Overview
| Attribute | Value |
|-----------|-------|
| **Purpose** | Detect VMs, sandboxes, automation tools |
| **Measurement** | Hardware clock jitter via `performance.now()` |
| **Frequency** | 1000 samples/second |
| **Detection Method** | KL Divergence & Shapiro-Wilk W test |
| **Implementation** | Rust → WebAssembly |

### Build Order: Step-by-Step

#### Step 1: Rust Project Setup

#### Step 2: Core Jitter Collection
```rust
// src/lib.rs
use wasm_bindgen::prelude::*;
use js_sys::Date;
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;

#[derive(Serialize, Deserialize, Clone)]
pub struct JitterSample {
    pub timestamp: f64,
    pub value: f64,  // Microseconds
}

#[wasm_bindgen]
pub struct ChronosEngine {
    samples: VecDeque<f64>,
    window_size: usize,
    baseline_mean: f64,
    baseline_std: f64,
}

#[wasm_bindgen]
impl ChronosEngine {
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            samples: VecDeque::with_capacity(1000),
            window_size: 1000,
            baseline_mean: 15.0,  // Expected jitter in microseconds
            baseline_std: 3.0,
        }
    }

    pub fn measure(&mut self) -> f64 {
        // Capture high-precision timing
        let t1 = Date::now();
        let t2 = Date::now();
        let jitter = t2 - t1;
        
        self.samples.push_back(jitter);
        if self.samples.len() > self.window_size {
            self.samples.pop_front();
        }
        
        jitter
    }

    pub fn analyze(&self) -> JsValue {
        if self.samples.len() < 10 {
            return serde_wasm_bindgen::to_value(&AnalysisResult {
                score: 0.5,
                status: "INSUFFICIENT_DATA".to_string(),
                mean_jitter: 0.0,
                std_jitter: 0.0,
                shapiro_w: 0.0,
                kl_divergence: 0.0,
            }).unwrap();
        }

        // Calculate statistics
        let mean = self.calculate_mean();
        let std = self.calculate_std(mean);
        let shapiro_w = self.calculate_shapiro_wilk();
        let kl_div = self.calculate_kl_divergence(mean, std);
        
        // Determine score
        let mut score = 1.0;
        if shapiro_w < 0.95 { score -= 0.5; }
        if kl_div > 0.5 { score -= 0.5; }
        if mean < 0.1 || mean > 50.0 { score -= 0.3; }
        score = score.max(0.0).min(1.0);
        
        let status = if score >= 0.8 { "CLEAR" } 
                     else if score >= 0.5 { "SUSPECT" } 
                     else { "ANOMALY" };

        serde_wasm_bindgen::to_value(&AnalysisResult {
            score: score,
            status: status.to_string(),
            mean_jitter: mean,
            std_jitter: std,
            shapiro_w: shapiro_w,
            kl_divergence: kl_div,
        }).unwrap()
    }

    fn calculate_mean(&self) -> f64 {
        self.samples.iter().sum::<f64>() / self.samples.len() as f64
    }

    fn calculate_std(&self, mean: f64) -> f64 {
        let variance = self.samples.iter()
            .map(|&x| (x - mean).powi(2))
            .sum::<f64>() / self.samples.len() as f64;
        variance.sqrt()
    }

    fn calculate_shapiro_wilk(&self) -> f64 {
        // Simplified Shapiro-Wilk W statistic approximation
        // For production, use a proper implementation
        let n = self.samples.len() as f64;
        let mean = self.calculate_mean();
        let std = self.calculate_std(mean);
        
        // Approximate W statistic
        let mut sorted: Vec<f64> = self.samples.iter().cloned().collect();
        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
        
        let sum_squared = sorted.iter()
            .map(|&x| (x - mean).powi(2))
            .sum::<f64>();
        
        // Simplified Shapiro-Francia approximation
        let w = 1.0 - (std / mean).abs().min(0.5);
        w.max(0.0).min(1.0)
    }

    fn calculate_kl_divergence(&self, mean: f64, std: f64) -> f64 {
        // KL Divergence between empirical and baseline normal distributions
        0.5 * (
            (self.baseline_std.powi(2) / std.powi(2)) +
            (std.powi(2) / self.baseline_std.powi(2)) -
            1.0 +
            ((mean - self.baseline_mean).powi(2) / self.baseline_std.powi(2))
        )
    }

    pub fn reset(&mut self) {
        self.samples.clear();
    }
}

#[derive(Serialize, Deserialize)]
struct AnalysisResult {
    score: f64,
    status: String,
    mean_jitter: f64,
    std_jitter: f64,
    shapiro_w: f64,
    kl_divergence: f64,
}
```

```rust
// src/lib.rs
use wasm_bindgen::prelude::*;
use web_sys::{AudioContext, AudioBuffer, AudioBufferSourceNode};
use std::f32::consts::PI;
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize)]
pub struct EchoResult {
    pub delay_ms: f32,
    pub status: String,
    pub confidence: f32,
    pub raw_delay_samples: f32,
}

#[wasm_bindgen]
pub struct EchoEngine {
    sample_rate: f32,
    chirp_duration: f32,
    chirp_buffer: Vec<f32>,
    audio_context: Option<AudioContext>,
}

#[wasm_bindgen]
impl EchoEngine {
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            sample_rate: 48000.0,
            chirp_duration: 0.5,
            chirp_buffer: Vec::new(),
            audio_context: None,
        }
    }

    pub fn generate_chirp(&mut self) {
        let num_samples = (self.sample_rate * self.chirp_duration) as usize;
        self.chirp_buffer = Vec::with_capacity(num_samples);
        
        let start_freq = 200.0;
        let end_freq = 4000.0;
        
        for n in 0..num_samples {
            let t = n as f32 / self.sample_rate;
            let freq = start_freq + (end_freq - start_freq) * t / self.chirp_duration;
            let phase = 2.0 * PI * freq * t;
            self.chirp_buffer.push(phase.sin());
        }
    }

    pub fn get_chirp_array(&self) -> js_sys::Float32Array {
        js_sys::Float32Array::from(&self.chirp_buffer[..])
    }

    pub fn cross_correlate(&self, recorded: &[f32]) -> EchoResult {
        // Ensure both signals are the same length
        let min_len = self.chirp_buffer.len().min(recorded.len());
        let chirp = &self.chirp_buffer[..min_len];
        let rec = &recorded[..min_len];
        
        // FFT-based cross-correlation
        let fft_size = min_len.next_power_of_two();
        
        // Pad signals to FFT size
        let mut chirp_fft = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        let mut rec_fft = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        
        for i in 0..min_len {
            chirp_fft[i] = num_complex::Complex32::new(chirp[i], 0.0);
            rec_fft[i] = num_complex::Complex32::new(rec[i], 0.0);
        }
        
        // FFT
        let mut planner = rustfft::FftPlanner::new();
        let fft = planner.plan_fft_forward(fft_size);
        let ifft = planner.plan_fft_inverse(fft_size);
        
        fft.process(&mut chirp_fft);
        fft.process(&mut rec_fft);
        
        // Cross-correlation in frequency domain
        let mut result = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        for i in 0..fft_size {
            result[i] = chirp_fft[i] * rec_fft[i].conj();
        }
        
        ifft.process(&mut result);
        
        // Find peak
        let mut max_val = 0.0;
        let mut max_idx = 0;
        for (i, val) in result.iter().enumerate() {
            let mag = val.norm();
            if mag > max_val {
                max_val = mag;
                max_idx = i;
            }
        }
        
        // Convert to delay in milliseconds
        let delay_samples = if max_idx > fft_size / 2 { 
            max_idx as f32 - fft_size as f32 
        } else { 
            max_idx as f32 
        };
        let delay_ms = delay_samples / self.sample_rate * 1000.0;
        
        // Determine status
        let (status, confidence) = if delay_ms > 6.0 {
            ("ANOMALY".to_string(), 0.95)
        } else if delay_ms < 1.0 {
            ("SUSPECT".to_string(), 0.70)
        } else {
            ("CLEAR".to_string(), 0.90)
        };
        
        EchoResult {
            delay_ms,
            status,
            confidence,
            raw_delay_samples: delay_samples,
        }
    }
}
```

```typescript
// echo-client.ts
import init, { EchoEngine } from './pkg/echo';

export class EchoClient {
  private engine: EchoEngine;
  private audioContext: AudioContext;
  private microphoneStream: MediaStream;
  private isRunning: boolean = false;
  
  async initialize() {
    await init();
    this.engine = new EchoEngine();
    this.audioContext = new AudioContext({ sampleRate: 48000 });
    
    // Request microphone
    this.microphoneStream = await navigator.mediaDevices.getUserMedia({
      audio: {
        sampleRate: 48000,
        channelCount: 1,
        echoCancellation: false,
        noiseSuppression: false,
        autoGainControl: false
      }
    });
  }
  
  async performMeasurement(): Promise<EchoResult> {
    // Generate chirp
    this.engine.generate_chirp();
    const chirpArray = this.engine.get_chirp_array();
    
    // Play chirp through speaker
    const chirpBuffer = this.audioContext.createBuffer(1, chirpArray.length, 48000);
    const channelData = chirpBuffer.getChannelData(0);
    for (let i = 0; i < chirpArray.length; i++) {
      channelData[i] = chirpArray[i];
    }
    
    const source = this.audioContext.createBufferSource();
    source.buffer = chirpBuffer;
    source.connect(this.audioContext.destination);
    
    // Record microphone
    const mediaStreamSource = this.audioContext.createMediaStreamSource(this.microphoneStream);
    const recorder = this.audioContext.createScriptProcessor(4096, 1, 1);
    const recordedSamples: Float32Array[] = [];
    
    recorder.onaudioprocess = (event) => {
      const input = event.inputBuffer.getChannelData(0);
      recordedSamples.push(new Float32Array(input));
    };
    
    mediaStreamSource.connect(recorder);
    recorder.connect(this.audioContext.destination);
    
    // Start playing and recording
    source.start();
    
    // Wait for recording
    await new Promise(resolve => setTimeout(resolve, 600));
    
    // Disconnect recording
    recorder.disconnect();
    source.disconnect();
    
    // Combine recorded samples
    let totalLength = 0;
    for (const chunk of recordedSamples) {
      totalLength += chunk.length;
    }
    const combined = new Float32Array(totalLength);
    let offset = 0;
    for (const chunk of recordedSamples) {
      combined.set(chunk, offset);
      offset += chunk.length;
    }
    
    // Run cross-correlation
    const result = this.engine.cross_correlate(Array.from(combined));
    return result;
  }
}
```

```rust
// src/lib.rs
use wasm_bindgen::prelude::*;
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;

#[derive(Serialize, Deserialize, Clone)]
pub struct EyeData {
    pub pupil_x: f32,
    pub pupil_y: f32,
    pub glint_x: f32,
    pub glint_y: f32,
    pub vector_x: f32,
    pub vector_y: f32,
}

#[derive(Serialize, Deserialize, Clone)]
pub struct IrisResult {
    pub score: f32,
    pub status: String,
    pub smoothness: f32,
    pub consistency: f32,
    pub trajectory_entropy: f32,
}

#[wasm_bindgen]
pub struct IrisEngine {
    trajectory_left: VecDeque<EyeData>,
    trajectory_right: VecDeque<EyeData>,
    window_size: usize,
    frame_count: u32,
}

#[wasm_bindgen]
impl IrisEngine {
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            trajectory_left: VecDeque::with_capacity(100),
            trajectory_right: VecDeque::with_capacity(100),
            window_size: 100,
            frame_count: 0,
        }
    }

    pub fn process_frame(&mut self, data: JsValue) -> IrisResult {
        self.frame_count += 1;
        
        // Parse face data from JavaScript
        let face_data: FaceData = serde_wasm_bindgen::from_value(data).unwrap();
        
        // Process left eye
        if let Some(left) = face_data.left_eye {
            let vector_x = left.glint_x - left.pupil_x;
            let vector_y = left.glint_y - left.pupil_y;
            self.trajectory_left.push_back(EyeData {
                pupil_x: left.pupil_x,
                pupil_y: left.pupil_y,
                glint_x: left.glint_x,
                glint_y: left.glint_y,
                vector_x,
                vector_y,
            });
        }
        
        // Process right eye
        if let Some(right) = face_data.right_eye {
            let vector_x = right.glint_x - right.pupil_x;
            let vector_y = right.glint_y - right.pupil_y;
            self.trajectory_right.push_back(EyeData {
                pupil_x: right.pupil_x,
                pupil_y: right.pupil_y,
                glint_x: right.glint_x,
                glint_y: right.glint_y,
                vector_x,
                vector_y,
            });
        }
        
        // Maintain window size
        if self.trajectory_left.len() > self.window_size {
            self.trajectory_left.pop_front();
        }
        if self.trajectory_right.len() > self.window_size {
            self.trajectory_right.pop_front();
        }
        
        // Calculate metrics
        let (smoothness, consistency, entropy) = self.calculate_metrics();
        
        // Score calculation
        let mut score = 1.0;
        if smoothness < 0.3 { score -= 0.5; }
        if consistency < 0.4 { score -= 0.3; }
        if entropy < 0.2 { score -= 0.4; }
        score = score.max(0.0).min(1.0);
        
        let status = if score >= 0.8 { "CLEAR" }
                     else if score >= 0.5 { "SUSPECT" }
                     else { "ANOMALY" };
        
        IrisResult {
            score,
            status: status.to_string(),
            smoothness,
            consistency,
            trajectory_entropy: entropy,
        }
    }

    fn calculate_metrics(&self) -> (f32, f32, f32) {
        // Combine both eyes for analysis
        let mut all_vectors: Vec<&EyeData> = Vec::new();
        all_vectors.extend(self.trajectory_left.iter());
        all_vectors.extend(self.trajectory_right.iter());
        
        if all_vectors.len() < 10 {
            return (0.5, 0.5, 0.5);
        }
        
        // Calculate smoothness (jerk)
        let mut accelerations = Vec::new();
        for i in 1..all_vectors.len() {
            let dx = all_vectors[i].vector_x - all_vectors[i-1].vector_x;
            let dy = all_vectors[i].vector_y - all_vectors[i-1].vector_y;
            let accel = (dx*dx + dy*dy).sqrt();
            accelerations.push(accel);
        }
        
        // Calculate jerk (change in acceleration)
        let mut jerks = Vec::new();
        for i in 1..accelerations.len() {
            let jerk = (accelerations[i] - accelerations[i-1]).abs();
            jerks.push(jerk);
        }
        
        let mean_jerk = if jerks.is_empty() { 0.0 } else {
            jerks.iter().sum::<f32>() / jerks.len() as f32
        };
        let std_jerk = if jerks.is_empty() { 0.0 } else {
            let variance = jerks.iter()
                .map(|&j| (j - mean_jerk).powi(2))
                .sum::<f32>() / jerks.len() as f32;
            variance.sqrt()
        };
        
        // Normal human movement has high entropy
        // Low std jerk = artificial (too smooth)
        let smoothness = if std_jerk > 0.1 { 1.0 } else { 0.2 };
        
        // Calculate consistency (vector length variation)
        let mut lengths = Vec::new();
        for data in all_vectors.iter() {
            let len = (data.vector_x*data.vector_x + data.vector_y*data.vector_y).sqrt();
            lengths.push(len);
        }
        let mean_len = lengths.iter().sum::<f32>() / lengths.len() as f32;
        let len_std = lengths.iter()
            .map(|&l| (l - mean_len).powi(2))
            .sum::<f32>() / lengths.len() as f32;
        let consistency = 1.0 - (len_std / (mean_len + 0.001)).min(1.0);
        
        // Entropy of trajectory
        let entropy = if all_vectors.len() > 20 {
            // Shannon entropy of vector directions
            let mut direction_counts = [0; 8];
            for data in all_vectors.iter() {
                let angle = data.vector_y.atan2(data.vector_x);
                let idx = ((angle + PI) / (2.0 * PI / 8.0)) as usize % 8;
                direction_counts[idx] += 1;
            }
            let probs: Vec<f32> = direction_counts.iter()
                .map(|&c| c as f32 / all_vectors.len() as f32)
                .collect();
            let mut entropy = 0.0;
            for &p in probs.iter() {
                if p > 0.0 {
                    entropy -= p * p.log2();
                }
            }
            entropy / 3.0 // Normalize to 0-1
        } else {
            0.5
        };
        
        (smoothness, consistency, entropy)
    }

    pub fn reset(&mut self) {
        self.trajectory_left.clear();
        self.trajectory_right.clear();
        self.frame_count = 0;
    }
}
```

```rust
// src/lib.rs
use wasm_bindgen::prelude::*;
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;

#[derive(Serialize, Deserialize)]
pub struct LipsyncResult {
    pub sync_score: f32,
    pub status: String,
    pub drift_ms: f32,
    pub coherence: f32,
}

#[wasm_bindgen]
pub struct LipsyncEngine {
    viseme_buffer: VecDeque<f32>,
    audio_energy_buffer: VecDeque<f32>,
    window_size: usize,
    sample_rate: f32,
}

#[wasm_bindgen]
impl LipsyncEngine {
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            viseme_buffer: VecDeque::with_capacity(300),
            audio_energy_buffer: VecDeque::with_capacity(300),
            window_size: 300, // 10 seconds at 30fps
            sample_rate: 30.0, // Frames per second
        }
    }

    pub fn process_frame(&mut self, viseme: f32, audio_energy: f32) -> LipsyncResult {
        // Store data
        self.viseme_buffer.push_back(viseme);
        self.audio_energy_buffer.push_back(audio_energy);
        
        if self.viseme_buffer.len() > self.window_size {
            self.viseme_buffer.pop_front();
            self.audio_energy_buffer.pop_front();
        }
        
        // Need minimum data for analysis
        if self.viseme_buffer.len() < 60 { // 2 seconds
            return LipsyncResult {
                sync_score: 0.5,
                status: "INSUFFICIENT_DATA".to_string(),
                drift_ms: 0.0,
                coherence: 0.0,
            };
        }
        
        // Perform cross-correlation
        let (sync_score, drift_samples, coherence) = self.calculate_sync();
        
        let drift_ms = drift_samples / self.sample_rate * 1000.0;
        
        // Determine status
        let status = if sync_score >= 0.85 {
            "CLEAR".to_string()
        } else if sync_score >= 0.70 {
            "SUSPECT".to_string()
        } else {
            "ANOMALY".to_string()
        };
        
        LipsyncResult {
            sync_score,
            status,
            drift_ms,
            coherence,
        }
    }

    fn calculate_sync(&self) -> (f32, f32, f32) {
        let n = self.viseme_buffer.len();
        
        // Convert to vectors for FFT
        let viseme_vec: Vec<f32> = self.viseme_buffer.iter().cloned().collect();
        let audio_vec: Vec<f32> = self.audio_energy_buffer.iter().cloned().collect();
        
        // FFT size (next power of two)
        let fft_size = n.next_power_of_two();
        
        // Prepare FFT data
        let mut viseme_fft = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        let mut audio_fft = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        
        for i in 0..n {
            viseme_fft[i] = num_complex::Complex32::new(viseme_vec[i], 0.0);
            audio_fft[i] = num_complex::Complex32::new(audio_vec[i], 0.0);
        }
        
        // FFT
        let mut planner = rustfft::FftPlanner::new();
        let fft = planner.plan_fft_forward(fft_size);
        let ifft = planner.plan_fft_inverse(fft_size);
        
        fft.process(&mut viseme_fft);
        fft.process(&mut audio_fft);
        
        // Cross-correlation
        let mut cross_corr = vec![num_complex::Complex32::new(0.0, 0.0); fft_size];
        for i in 0..fft_size {
            cross_corr[i] = viseme_fft[i] * audio_fft[i].conj();
        }
        
        ifft.process(&mut cross_corr);
        
        // Find peak
        let mut max_val = 0.0;
        let mut max_idx = 0;
        for (i, val) in cross_corr.iter().enumerate() {
            let mag = val.norm();
            if mag > max_val {
                max_val = mag;
                max_idx = i;
            }
        }
        
        // Calculate drift
        let drift_samples = if max_idx > fft_size / 2 {
            max_idx as f32 - fft_size as f32
        } else {
            max_idx as f32
        };
        
        // Calculate coherence (power ratio)
        let total_power: f32 = cross_corr.iter()
            .map(|c| c.norm())
            .sum();
        let peak_power = max_val;
        let coherence = if total_power > 0.0 { peak_power / total_power } else { 0.0 };
        
        // Sync score (normalized)
        let sync_score = if max_idx == 0 {
            1.0 // Perfect sync
        } else {
            1.0 - (drift_samples.abs() / (fft_size as f32 / 2.0))
        }.max(0.0).min(1.0);
        
        (sync_score, drift_samples, coherence)
    }
```

```python
    async def process_audio(self, audio_chunk: np.ndarray) -> Dict[str, float]:
        """Process audio chunk for voice stress"""
        # Resample to 16kHz for model
        audio_resampled = librosa.resample(audio_chunk, orig_sr=48000, target_sr=16000)
        
        # Extract features
        f0, voiced_flag, voiced_probs = librosa.pyin(
            audio_resampled,
            fmin=50,
            fmax=400,
            sr=16000
        )

        voiced_f0 = f0[voiced_flag]
        if len(voiced_f0) > 1:
            jitter = np.std(voiced_f0) / np.mean(voiced_f0)
        else:
            jitter = 0.1
        
        # Calculate shimmer (amplitude variation)
        rms = librosa.feature.rms(y=audio_resampled)[0]
        shimmer = np.std(rms) / np.mean(rms) if np.mean(rms) > 0 else 0.1
        
        # Calculate harmonics-to-noise ratio
        harmonic = librosa.effects.harmonic(audio_resampled)
        hnr_ratio = np.mean(harmonic) / np.mean(audio_resampled) if np.mean(audio_resampled) > 0 else 1.0
        
        # Stress score (0-1)
        stress_score = (jitter * 0.4 + shimmer * 0.3 + (1 - hnr_ratio) * 0.3)
        stress_score = min(1.0, stress_score)
```

```python
        """Extract pupil and glint positions from face landmarks"""

        LEFT_EYE_INDICES = [33, 133, 157, 158, 159, 160, 161, 173, 246]

        right_eye_points = []
        for idx in RIGHT_EYE_INDICES:
            landmark = landmarks.landmark[idx]
            x = int(landmark.x * frame_width)
            y = int(landmark.y * frame_height)
            right_eye_points.append((x, y))
        
        # Calculate pupil position (center of eye region)
        left_pupil_x = np.mean([p[0] for p in left_eye_points])
        left_pupil_y = np.mean([p[1] for p in left_eye_points])
        right_pupil_x = np.mean([p[0] for p in right_eye_points])
        right_pupil_y = np.mean([p[1] for p in right_eye_points])
        
        # Glint detection (brightest point in eye region)
        # Simplified - using a corner of the eye
        left_glint_x = left_eye_points[0][0] + 10
        left_glint_y = left_eye_points[0][1] - 5
        right_glint_x = right_eye_points[0][0] + 10
        right_glint_y = right_eye_points[0][1] - 5

            'left': EyeData(

                pupil_y=left_pupil_y,
                glint_x=left_glint_x,
                glint_y=left_glint_y,
                vector_x=left_glint_x - left_pupil_x,

    def calculate_ear(self, eye_points):

        # Vertical distances

        v2 = np.linalg.norm(np.array(eye_points[2]) - np.array(eye_points[4]))
        # Horizontal distance
        h = np.linalg.norm(np.array(eye_points[0]) - np.array(eye_points[3]))
        # EAR formula
        ear = (v1 + v2) / (2.0 * h)
        return ear
```

```python
                    channel='orchestrator:control'
                )
                if control and control['type'] == 'message':
                    await self.handle_control(control)
                    
            except Exception as e:
                print(f"Orchestrator error: {e}")
                await asyncio.sleep(0.1)
    
    async def process_agent_output(self, data: Dict[str, Any]):
        """Process output from an agent"""
        session_id = data.get('session_id')
        agent = data.get('agent')
        result = data.get('result')
        timestamp = data.get('timestamp', datetime.now().timestamp())
        
        # Get or create session
        if session_id not in self.sessions:
            self.sessions[session_id] = SessionState(session_id=session_id)
        
        session = self.sessions[session_id]
        session.agent_outputs[agent] = result
        session.last_update = timestamp
        
        # Calculate trust score
        trust_score = self.calculate_trust_score(session)
        
        # Update trust history
        session.trust_history.append({
            'timestamp': timestamp,
            'score': trust_score,
            'agent': agent
        })
        
        # Generate verdict if enough data
        if len(session.trust_history) >= 10:
            verdict = self.generate_verdict(session)
            session.verdict = verdict
        
        # Publish to dashboard
        await self.publish_dashboard_update(session_id, trust_score, session)
        
        # Store in database
        await self.store_session_data(session)
    
    def calculate_trust_score(self, session: SessionState) -> float:
        """Calculate Bayesian trust score from agent outputs"""
        outputs = session.agent_outputs
        
        # Extract scores from each agent
        scores = {}
        for agent, output in outputs.items():
            if agent in self.agent_weights:
                if isinstance(output, dict):
                    # Try different score fields
                    score = output.get('score', output.get('trust_score', output.get('verification_score', 0.5)))
                    # Handle status-based scoring
                    if 'status' in output:
                        status_score = {
                            'CLEAR': 1.0,
                            'SUSPECT': 0.5,
                            'ANOMALY': 0.0,
                            'FRAUD_DETECTED': 0.0,
                            'INSUFFICIENT_DATA': 0.5,
                            'NO_FACE': 0.3
                        }.get(output['status'], 0.5)
                        # Use the lower of score and status_score
                        score = min(score, status_score)
                    scores[agent] = score
        
        if not scores:
            return 0.5
        
        # Weighted sum
        weighted_sum = sum(
            scores.get(agent, 0.5) * self.agent_weights.get(agent, 0.1)
            for agent in self.agent_weights.keys()
        )
        total_weight = sum(self.agent_weights.values())
        
        # Bayesian update
        alpha = self.prior_alpha + weighted_sum
        beta_val = self.prior_beta + (total_weight - weighted_sum)
        
        trust_score = alpha / (alpha + beta_val)
        
        # Apply anomaly penalty if critical anomalies detected
        anomaly_count = self.count_anomalies(session)
        if anomaly_count >= 2:
            trust_score *= (1.0 - anomaly_count * 0.1)
        
        return round(max(0.0, min(1.0, trust_score)), 3)
    
    def count_anomalies(self, session: SessionState) -> int:
        """Count critical anomalies from agent outputs"""
        anomaly_count = 0
        for agent, output in session.agent_outputs.items():
            if isinstance(output, dict):
                if output.get('status') == 'ANOMALY':
                    anomaly_count += 1
                if output.get('status') == 'FRAUD_DETECTED':
                    anomaly_count += 2
        return anomaly_count
    
    def generate_verdict(self, session: SessionState) -> Dict[str, Any]:
        """Generate verdict based on trust score and anomalies"""
        trust_history = session.trust_history
        if not trust_history:
            return {
                'status': 'PENDING',
                'recommendation': 'Insufficient data',
                'trust_score': 0.5,
                'anomalies': []
            }
        
        # Calculate recent average
        recent_scores = [h['score'] for h in trust_history[-10:]]
        recent_avg = np.mean(recent_scores) if recent_scores else 0.5
        
        # Check anomalies
        anomaly_count = self.count_anomalies(session)
        anomaly_details = self.get_anomaly_details(session)
        
        # Determine status
        if anomaly_count >= 2:
            status = 'FRAUD_DETECTED'
            recommendation = 'Immediate session termination recommended'
        elif recent_avg < self.thresholds['fraud']:
            status = 'FRAUD_DETECTED'
            recommendation = 'Trust score critically low. Investigate immediately.'
        elif recent_avg < self.thresholds['high_suspicion']:
            status = 'HIGH_SUSPICION'
            recommendation = 'Multiple red flags. Deep probe recommended.'
        elif recent_avg < self.thresholds['moderate_concern']:
            status = 'MODERATE_CONCERN'
            recommendation = 'Heightened scrutiny required. Continue assessment.'
        elif recent_avg >= self.thresholds['clear']:
            status = 'CLEAR'
            recommendation = 'Candidate appears authentic. Recommend positive evaluation.'
        else:
            status = 'PENDING'
            recommendation = 'Continue assessment for more data'
        
        # Calculate trend
        if len(trust_history) >= 20:
            old_avg = np.mean([h['score'] for h in trust_history[-20:-10]])
            trend = 'improving' if recent_avg > old_avg else 'declining'
        else:
            trend = 'stable'
        
        return {
            'status': status,
            'recommendation': recommendation,
            'trust_score': round(recent_avg, 3),
            'trend': trend,
            'anomalies': anomaly_details,
            'total_anomalies': anomaly_count,
            'timestamp': datetime.now().isoformat()
        }
    
    def get_anomaly_details(self, session: SessionState) -> List[Dict[str, Any]]:
        """Get detailed anomaly information"""
        anomalies = []
        for agent, output in session.agent_outputs.items():
            if isinstance(output, dict):
                if output.get('status') in ['ANOMALY', 'FRAUD_DETECTED']:
                    anomalies.append({
                        'agent': agent.upper(),
                        'status': output.get('status'),
                        'score': output.get('score', 0.0),
                        'message': output.get('message', f'{agent.upper()} anomaly detected')
                    })
        return anomalies
    
    async def publish_dashboard_update(self, session_id: str, trust_score: float, session: SessionState):

    def get_agent_status(self, session: SessionState) -> Dict[str, str]:
```

```typescript
// src/hooks/useWebRTC.ts
import { useState, useRef, useEffect } from 'react';
import Peer from 'simple-peer';
import { useWebSocket } from './useWebSocket';

export const useWebRTC = (
  remoteVideoRef: React.RefObject<HTMLVideoElement>,
  ws: ReturnType<typeof useWebSocket>
) => {
  const [isMuted, setIsMuted] = useState(false);
  const [isVideoOff, setIsVideoOff] = useState(false);
  const [isSharing, setIsSharing] = useState(false);
  
  const localVideoRef = useRef<HTMLVideoElement>(null);
  const peerRef = useRef<Peer.Instance | null>(null);
  const streamRef = useRef<MediaStream | null>(null);
  
  useEffect(() => {
    // Initialize media
    navigator.mediaDevices.getUserMedia({
      video: { width: 1280, height: 720, frameRate: 30 },
      audio: { sampleRate: 48000, echoCancellation: true }
    }).then(stream => {
      streamRef.current = stream;
      if (localVideoRef.current) {
        localVideoRef.current.srcObject = stream;
      }
    });
    
    return () => {
      streamRef.current?.getTracks().forEach(t => t.stop());
      peerRef.current?.destroy();
    };
  }, []);
  
  useEffect(() => {
    // Handle WebRTC signaling
    if (!ws.socket) return;
    
    ws.socket.on('webrtc-offer', ({ offer, from }) => {
      if (!streamRef.current) return;
      
      peerRef.current = new Peer({
        initiator: false,
        trickle: false,
        stream: streamRef.current
      });
      
      peerRef.current.on('signal', (signal) => {
        ws.socket?.emit('webrtc-answer', {
          targetId: from,
          answer: signal
        });
      });
      
      peerRef.current.on('stream', (remoteStream) => {
        if (remoteVideoRef.current) {
          remoteVideoRef.current.srcObject = remoteStream;
        }
      });
      
      peerRef.current.signal(offer);
    });
    
    ws.socket.on('webrtc-answer', ({ answer, from }) => {
      peerRef.current?.signal(answer);
    });
    
    ws.socket.on('ice-candidate', ({ candidate }) => {
      peerRef.current?.signal(candidate);
    });
    
  }, [ws.socket, remoteVideoRef]);
  
  const createOffer = () => {
    if (!streamRef.current) return;
    
    peerRef.current = new Peer({
      initiator: true,
      trickle: false,
      stream: streamRef.current
    });
    
    peerRef.current.on('signal', (signal) => {
      ws.socket?.emit('webrtc-offer', {
        targetId: 'interviewer',
        offer: signal
      });
    });
    
    peerRef.current.on('stream', (remoteStream) => {
      if (remoteVideoRef.current) {
        remoteVideoRef.current.srcObject = remoteStream;
      }
    });
  };
  
  const toggleMute = () => {
    if (!streamRef.current) return;
    const audioTrack = streamRef.current.getAudioTracks()[0];
    if (audioTrack) {
      audioTrack.enabled = !audioTrack.enabled;
      setIsMuted(!audioTrack.enabled);
    }
  };
  
  const toggleVideo = () => {
    if (!streamRef.current) return;
    const videoTrack = streamRef.current.getVideoTracks()[0];
    if (videoTrack) {
      videoTrack.enabled = !videoTrack.enabled;
      setIsVideoOff(!videoTrack.enabled);
    }
  };
  
  const toggleScreenShare = async () => {
    if (isSharing) {
      // Stop sharing
      streamRef.current?.getVideoTracks().forEach(t => t.stop());
      // Restore camera
      const cameraStream = await navigator.mediaDevices.getUserMedia({
        video: { width: 1280, height: 720, frameRate: 30 }
      });
      const videoTrack = cameraStream.getVideoTracks()[0];
      streamRef.current?.addTrack(videoTrack);
      setIsSharing(false);
    } else {
      // Start sharing
      const screenStream = await navigator.mediaDevices.getDisplayMedia({
        video: { width: 1920, height: 1080 }
      });
      const videoTrack = screenStream.getVideoTracks()[0];
      // Replace track
      const oldTrack = streamRef.current?.getVideoTracks()[0];
      if (oldTrack) {
        streamRef.current?.removeTrack(oldTrack);
        oldTrack.stop();
      }
      streamRef.current?.addTrack(videoTrack);
      setIsSharing(true);
    }
  };
  
  return {
    localVideoRef,
    isMuted,
    isVideoOff,
    isSharing,
    toggleMute,
    toggleVideo,
    toggleScreenShare,
    createOffer
  };
};
```

```python
# encryption.py
from cryptography.fernet import Fernet
import base64
import hashlib

class DataEncryption:
    def __init__(self, key: bytes):
        self.cipher = Fernet(key)
    
    @classmethod
    def generate_key(cls) -> bytes:
        return Fernet.generate_key()
    
    def encrypt(self, data: bytes) -> bytes:
        return self.cipher.encrypt(data)
    
    def decrypt(self, encrypted_data: bytes) -> bytes:
        return self.cipher.decrypt(encrypted_data)

class ZeroPIIFilter:
    """Zero-PII compliance filter"""
    
    @staticmethod
    def anonymize(data: dict) -> dict:
        """Remove all personally identifiable information"""
        sensitive_fields = [
            'video', 'audio', 'face_data', 'name', 'email',
            'phone', 'address', 'ssn', 'id_number'
        ]
        
        result = data.copy()
        for field in sensitive_fields:
            if field in result:
                if field in ['video', 'audio', 'face_data']:
                    result[field] = None
                else:
                    result[field] = 'REDACTED'
        return result
    
    @staticmethod
    def hash_identifier(identifier: str) -> str:
        """Hash any identifier for tracking without storing raw"""
        return hashlib.sha256(identifier.encode()).hexdigest()
    
    @staticmethod
    def vectorize_sensitive_data(data: dict) -> dict:
        """Convert sensitive data to mathematical vectors only"""
        # For frame timing data
        if 'timestamps' in data:
            data['timestamps'] = data['timestamps'][:100]  # Only sample
        return data
```

```python
# middleware.py
from fastapi import Request, HTTPException
import jwt
import time
from typing import Optional

class SecurityMiddleware:
    def __init__(self, secret_key: str):
        self.secret_key = secret_key
    
    async def validate_token(self, request: Request) -> Optional[str]:
        """Validate JWT token"""
        auth_header = request.headers.get('Authorization')
        if not auth_header or not auth_header.startswith('Bearer '):
            raise HTTPException(status_code=401, detail="Missing token")
        
        token = auth_header.split(' ')[1]
        try:
            payload = jwt.decode(token, self.secret_key, algorithms=['HS256'])
            return payload.get('session_id')
        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=401, detail="Token expired")
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=401, detail="Invalid token")
    
    async def rate_limit(self, request: Request, max_requests: int = 100, window: int = 60):
        """Rate limiting middleware"""
        # Implementation using Redis counter
        pass
    
    def create_session_token(self, session_id: str) -> str:
        """Create a session token"""
        payload = {
            'session_id': session_id,
            'exp': time.time() + 3600  # 1 hour
        }
        return jwt.encode(payload, self.secret_key, algorithm='HS256')
```

```yaml
# docker-compose.yml
version: '3.8'

services:
  # Redis
  redis:
    image: redis:7-alpine
    container_name: aegis-redis
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    networks:
      - aegis-network

  # PostgreSQL
  postgres:
    image: postgres:15
    container_name: aegis-postgres
    environment:
      POSTGRES_DB: aegis
      POSTGRES_USER: aegis
      POSTGRES_PASSWORD: ${PG_PASSWORD}
    ports:
      - "5432:5432"
    volumes:
      - postgres-data:/var/lib/postgresql/data
    networks:
      - aegis-network

  # TimescaleDB
  timescaledb:
    image: timescale/timescaledb:latest-pg15
    container_name: aegis-timescale
    environment:
      POSTGRES_DB: aegis_metrics
      POSTGRES_USER: aegis
      POSTGRES_PASSWORD: ${TIMESCALE_PASSWORD}
    ports:
      - "5433:5432"
    volumes:
      - timescale-data:/var/lib/postgresql/data
    networks:
      - aegis-network

  # Neo4j
  neo4j:
    image: neo4j:5
    container_name: aegis-neo4j
    environment:
      NEO4J_AUTH: neo4j/${NEO4J_PASSWORD}
    ports:
      - "7474:7474"
      - "7687:7687"
    volumes:
      - neo4j-data:/data
    networks:
      - aegis-network

  # Nginx
  nginx:
    image: nginx:alpine
    container_name: aegis-nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
    depends_on:
      - redis
      - postgres
      - timescaledb
      - neo4j
    networks:
      - aegis-network

networks:
  aegis-network:
    driver: bridge

volumes:
  redis-data:
  postgres-data:
  timescale-data:
  neo4j-data:
```

```rust
// packages/chronos/src/lib.rs
use wasm_bindgen::prelude::*;
use js_sys::Date;
use std::collections::VecDeque;
use serde::{Deserialize, Serialize};

const WINDOW_SIZE: usize = 1000;

#[wasm_bindgen]
pub struct ChronosEngine {
    samples: VecDeque<f64>,
}

#[wasm_bindgen]
impl ChronosEngine {
    #[wasm_bindgen(constructor)]
    pub fn new() -> Self {
        Self {
            samples: VecDeque::with_capacity(WINDOW_SIZE),
        }
    }
    
    /// Measure a single timing sample
    /// Returns the jitter value in microseconds
    pub fn measure(&mut self) -> f64 {
        let t1 = Date::now();
        let t2 = Date::now();
        let jitter = t2 - t1;
        
        self.samples.push_back(jitter);
        if self.samples.len() > WINDOW_SIZE {
            self.samples.pop_front();
        }
        
        jitter
    }
    
    /// Get all current samples (for testing)
    pub fn get_samples(&self) -> Vec<f64> {
        self.samples.iter().cloned().collect()
    }
    
    /// Get sample count
    pub fn sample_count(&self) -> usize {
        self.samples.len()
    }
    
    /// Clear all samples
    pub fn clear(&mut self) {
        self.samples.clear();
    }
}

#[wasm_bindgen]
pub fn measure_batch(count: usize) -> Vec<f64> {
    let mut engine = ChronosEngine::new();
    let mut results = Vec::with_capacity(count);
    
    for _ in 0..count {
        results.push(engine.measure());
    }
    
    results
}
```

### Implementation Steps

4. Implement `calculate_kl_divergence()` method

### Code Requirements
```rust
// packages/chronos/src/analysis.rs
use std::collections::VecDeque;

const BASELINE_MEAN: f64 = 15.0; // Expected jitter in microseconds
const BASELINE_STD: f64 = 3.0;

pub struct ChronosAnalyzer {
    samples: VecDeque<f64>,
}

impl ChronosAnalyzer {
    pub fn new(samples: VecDeque<f64>) -> Self {
        Self { samples }
    }
    
    pub fn calculate_mean(&self) -> f64 {
        if self.samples.is_empty() {
            return 0.0;
        }
        self.samples.iter().sum::<f64>() / self.samples.len() as f64
    }
    
    pub fn calculate_std(&self, mean: f64) -> f64 {
        if self.samples.is_empty() {
            return 0.0;
        }
        let variance = self.samples.iter()
            .map(|&x| (x - mean).powi(2))
            .sum::<f64>() / self.samples.len() as f64;
        variance.sqrt()
    }
    
    pub fn calculate_shapiro_wilk(&self) -> f64 {
        let n = self.samples.len() as f64;
        if n < 3.0 {
            return 1.0;
        }
        
        let mean = self.calculate_mean();
        let sorted: Vec<f64> = {
            let mut v: Vec<f64> = self.samples.iter().cloned().collect();
            v.sort_by(|a, b| a.partial_cmp(b).unwrap());
            v
        };
        
        // Calculate W statistic using Shapiro-Francia approximation
        // This is a simplified version for WASM performance
        let numerator: f64 = sorted.iter()
            .enumerate()
            .map(|(i, &x)| {
                let expected = (i as f64 + 0.5) / n;
                let z_score = stats::normal::inv_cdf(expected);
                z_score * x
            })
            .sum::<f64>().powi(2);
        
        let denominator: f64 = sorted.iter()
            .map(|&x| (x - mean).powi(2))
            .sum();
        
        if denominator < 1e-10 {
            return 1.0;
        }
        (numerator / denominator).max(0.0).min(1.0)
    }
    
    pub fn calculate_kl_divergence(&self, mean: f64, std: f64) -> f64 {
        // KL Divergence between empirical and baseline normal distributions
        // D_KL(P||Q) = log(σ2/σ1) + (σ1^2 + (μ1-μ2)^2)/(2*σ2^2) - 0.5
        let sigma1 = std;
        let sigma2 = BASELINE_STD;
        let mu1 = mean;
        let mu2 = BASELINE_MEAN;
        
        if sigma1 < 1e-10 || sigma2 < 1e-10 {
            return 1.0;
        }
        
        let term1 = (sigma2 / sigma1).ln();
        let term2 = (sigma1.powi(2) + (mu1 - mu2).powi(2)) / (2.0 * sigma2.powi(2));
        let result = term1 + term2 - 0.5;
        result.max(0.0)
    }
    
    pub fn analyze(&self) -> ChronosAnalysis {
        let mean = self.calculate_mean();
        let std = self.calculate_std(mean);
        let shapiro_w = self.calculate_shapiro_wilk();
        let kl_divergence = self.calculate_kl_divergence(mean, std);
        
        ChronosAnalysis {
            mean_jitter: mean,
            std_jitter: std,
            shapiro_w,
            kl_divergence,
            sample_count: self.samples.len(),
        }
    }
}

pub struct ChronosAnalysis {
    pub mean_jitter: f64,
    pub std_jitter: f64,
    pub shapiro_w: f64,
    pub kl_divergence: f64,
    pub sample_count: usize,
}
```

```rust
// packages/chronos/src/lib.rs (continued)
use crate::analysis::{ChronosAnalyzer, ChronosAnalysis};

#[derive(Serialize, Deserialize)]
pub struct ChronosResult {
    pub score: f64,
    pub status: String,
    pub mean_jitter: f64,
    pub std_jitter: f64,
    pub shapiro_w: f64,
    pub kl_divergence: f64,
    pub sample_count: usize,
}

const THRESHOLD_CLEAR: f64 = 0.8;
const THRESHOLD_SUSPECT: f64 = 0.5;

impl ChronosEngine {
    pub fn analyze(&self) -> ChronosResult {
        if self.samples.len() < 10 {
            return ChronosResult {
                score: 0.5,
                status: "INSUFFICIENT_DATA".to_string(),
                mean_jitter: 0.0,
                std_jitter: 0.0,
                shapiro_w: 0.0,
                kl_divergence: 0.0,
                sample_count: self.samples.len(),
            };
        }
        
        let analyzer = ChronosAnalyzer::new(self.samples.clone());
        let analysis = analyzer.analyze();
        
        let score = self.calculate_score(&analysis);
        let status = self.determine_status(score);
        
        ChronosResult {
            score,
            status,
            mean_jitter: analysis.mean_jitter,
            std_jitter: analysis.std_jitter,
            shapiro_w: analysis.shapiro_w,
            kl_divergence: analysis.kl_divergence,
            sample_count: self.samples.len(),
        }
    }
    
    fn calculate_score(&self, analysis: &ChronosAnalysis) -> f64 {
        let mut score = 1.0;
        
        // Shapiro-Wilk W statistic penalty
        if analysis.shapiro_w < 0.95 {
            score -= 0.5;
        }
        
        // KL Divergence penalty
        if analysis.kl_divergence > 0.5 {
            score -= 0.5;
        }
        
        // Mean jitter range penalty
        if analysis.mean_jitter < 0.1 || analysis.mean_jitter > 50.0 {
            score -= 0.3;
        }
        
        // Standard deviation penalty
        if analysis.std_jitter < 0.5 || analysis.std_jitter > 10.0 {
            score -= 0.2;
        }
        
        // Sample count penalty
        if analysis.sample_count < 50 {
            score -= 0.1;
        }
        
        score.max(0.0).min(1.0)
    }
    
    fn determine_status(&self, score: f64) -> String {
        if score >= THRESHOLD_CLEAR {
            "CLEAR".to_string()
        } else if score >= THRESHOLD_SUSPECT {
            "SUSPECT".to_string()
        } else {
            "ANOMALY".to_string()
        }
    }
}
```
